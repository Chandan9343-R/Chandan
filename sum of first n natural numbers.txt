#include <stdio.h>
int main()
{
    int num, count, sum = 0;
    
    printf("Enter a positive integer: ");
    scanf("%d", &num); 
    
    for (count = 1; count <= num; ++count) // Loop starts
    {
        sum += count; // <--- The action to be repeated in the loop
    } // Loop ends
    
    printf("Sum = %d", sum);
    return 0;
}
